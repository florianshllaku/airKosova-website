# Speed / Headless Options (Playwright)

# Force headless mode regardless of Windows defaults / UI checkbox.

# Enable a few safe speed tweaks (tighter timeouts + shorter post-click delay).
# Default: true when headless, false when headed.
SCRAPER_FAST_MODE=true

# Block heavy resource types to speed up page load.
# Default: true when SCRAPER_FAST_MODE=true
SCRAPER_BLOCK_RESOURCES=true

# Reuse a single Chromium instance across searches (reduces launch overhead).
# Default: true
SCRAPER_REUSE_BROWSER=true

# Keep a warm page ready between searches (recommended for max speed between searches).
# This keeps a single Playwright page open and navigated back to the home form after scraping.
SCRAPER_KEEP_READY=true

# Reuse the same page for all searches (serialized; only 1 search at a time).
SCRAPER_REUSE_PAGE=true
